on_actions = {
	on_startup = {
		effect = {
			if = {
                limit = {
                    has_game_rule = {
                        rule = BHOI_government_policies
                        option = LIMITED
                    }
                }
                set_variable = { global.bhoi_government_limited = 1 }
            }
			if = {
                limit = {
                    has_game_rule = {
                        rule = BHOI_government_policies
                        option = UNLIMITED
                    }
                }
                set_variable = { global.bhoi_government_limited = 0 }
            }
		}
	}
    on_annex = {
        effect = {
            if = {
                limit = {
                    has_template = "Mercenaries"
                }
                delete_unit_template_and_units = {
                    division_template = "Mercenaries"
                }
            }
        }
    }
}
